// ── Toast ──
function toast(msg, type = 'info', duration = 2200) {
  const el = document.createElement('div');
  el.className = `toast ${type}`;
  el.textContent = msg;
  $('toast-container').appendChild(el);
  setTimeout(() => {
    el.style.animation = 'toastOut 0.25s cubic-bezier(.4,0,.2,1) forwards';
    setTimeout(() => el.remove(), 260);
  }, duration);
}

// ── Tab Management ──
function generateId() { return '_' + Math.random().toString(36).slice(2, 9); }

function createTab(name, content = '') {
  const id = generateId();
  const tab = { id, name, content, saved: true };
  tabs.push(tab);
  renderTabs();
  switchTab(id);
  return id;
}

function newUntitled(content = '') {
  untitledCount++;
  return createTab(`Untitled-${untitledCount}`, content);
}

function renderTabs() {
  // Remove all tab elements (keep new-tab-btn)
  [...tabsBar.querySelectorAll('.tab')].forEach(t => t.remove());
  tabs.forEach(tab => {
    const el = document.createElement('div');
    el.className = 'tab' + (tab.id === activeTabId ? ' active' : '') + (!tab.saved ? ' unsaved' : '');
    el.dataset.id = tab.id;
    el.innerHTML = `
      <span class="tab-dot"></span>
      <span class="tab-name" title="${escapeHtml(tab.name)}">${escapeHtml(tab.name)}</span>
      <span class="tab-close" data-close="${tab.id}">✕</span>
    `;
    tabsBar.insertBefore(el, newTabBtn);
  });
}

function switchTab(id) {
  // Save current content before switching
  if (activeTabId) {
    const cur = tabs.find(t => t.id === activeTabId);
    if (cur) cur.content = editor.value;
  }
  activeTabId = id;
  const tab = tabs.find(t => t.id === id);
  if (!tab) return;
  editor.value = tab.content;
  renderTabs();
  updateStats();
  updateLineNumbers();
  editor.focus();
}

function closeTab(id) {
  const tab = tabs.find(t => t.id === id);
  if (!tab) return;
  if (!tab.saved) {
    if (!confirm(`"${tab.name}" has unsaved changes. Close anyway?`)) return;
  }
  const idx = tabs.findIndex(t => t.id === id);
  tabs.splice(idx, 1);
  if (tabs.length === 0) { newUntitled(); return; }
  const newActive = tabs[Math.min(idx, tabs.length - 1)].id;
  activeTabId = null;
  switchTab(newActive);
}

tabsBar.addEventListener('click', e => {
  const closeId = e.target.dataset.close;
  if (closeId) { closeTab(closeId); return; }
  const tabEl = e.target.closest('.tab');
  if (tabEl && tabEl.dataset.id !== activeTabId) switchTab(tabEl.dataset.id);
});
newTabBtn.addEventListener('click', () => newUntitled());

  newUntitled();
  toast('New file created', 'info');
}

function openFile() {
  $('file-input').click();
}

$('file-input').addEventListener('change', e => {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    createTab(file.name, ev.target.result);
    markSaved();
    toast(`Opened: ${file.name}`, 'success');
  };
  reader.readAsText(file);
  e.target.value = '';
});

function saveFile() {
  const tab = tabs.find(t => t.id === activeTabId);
  if (!tab) return;
  tab.content = editor.value;
  const blob = new Blob([tab.content], { type: 'text/plain' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = tab.name.includes('.') ? tab.name : tab.name + '.txt';
  a.click();
  URL.revokeObjectURL(a.href);
  markSaved(true);
  toast(`Saved: ${tab.name}`, 'success');
}

function markSaved(saved = true) {
  const tab = tabs.find(t => t.id === activeTabId);
  if (!tab) return;
  tab.saved = saved;
  renderTabs();
}

function printFile() {
  const tab = tabs.find(t => t.id === activeTabId);
  const content = editor.value;
  const w = window.open('', '_blank');
  w.document.write(`<pre style="font-family:monospace;font-size:13px;white-space:pre-wrap;">${escapeHtml(content)}</pre>`);
  w.document.close();
  w.print();
  toast('Sent to printer', 'info');
}

editor.addEventListener('keydown', e => {
  // Tab key → insert spaces
  if (e.key === 'Tab') {
    e.preventDefault();
    const s = editor.selectionStart, end = editor.selectionEnd;
    editor.value = editor.value.substring(0, s) + '  ' + editor.value.substring(end);
    editor.selectionStart = editor.selectionEnd = s + 2;
    const tab = tabs.find(t => t.id === activeTabId);
    if (tab) tab.content = editor.value;
  }
});

function updateCursor() {
  const val = editor.value;
  const pos = editor.selectionStart;
  const before = val.substring(0, pos);
  const line = before.split('\n').length;
  const col = before.split('\n').pop().length + 1;
  $('st-lines').textContent = `Ln ${line}, Col ${col}`;
}

// ── Stats ──
function updateStats() {
  const text = editor.value;
  const words = text.trim() === '' ? 0 : text.trim().split(/\s+/).length;
  const chars = text.length;
  const readTime = Math.max(1, Math.round(words / 200));
  $('st-words').textContent = `${words.toLocaleString()} word${words !== 1 ? 's' : ''}`;
  $('st-chars').textContent = `${chars.toLocaleString()} char${chars !== 1 ? 's' : ''}`;
  $('st-read').textContent = `~${readTime} min read`;
}

// ── Line Numbers ──
function updateLineNumbers() {
  if (!lineNumbersVisible) return;
  const lines = editor.value.split('\n').length;
  lineNums.innerHTML = Array.from({ length: lines }, (_, i) => `<div>${i + 1}</div>`).join('');
  lineNums.scrollTop = editor.scrollTop;
}

function toggleFind() {
  findBar.classList.toggle('open');

  } else {

$('btn-replace-one').addEventListener('click', () => {
  const term = findInput.value;
  const rep = replaceInput.value;
  if (!term || findMatches.length === 0) return;
  const pos = findMatches[findIndex < 0 ? 0 : findIndex % findMatches.length];
  const val = editor.value;
  editor.value = val.substring(0, pos) + rep + val.substring(pos + term.length);
  const tab = tabs.find(t => t.id === activeTabId);
  if (tab) { tab.content = editor.value; tab.saved = false; renderTabs(); }
  doFind();
  toast(`Replaced 1 match`, 'success');
});

$('btn-replace-all').addEventListener('click', () => {
  const term = findInput.value;
  const rep = replaceInput.value;
  if (!term) return;
  const count = (editor.value.split(term).length - 1);
  editor.value = editor.value.split(term).join(rep);
  const tab = tabs.find(t => t.id === activeTabId);
  if (tab) { tab.content = editor.value; tab.saved = false; renderTabs(); }
  doFind();
  toast(`Replaced ${count} match${count !== 1 ? 'es' : ''}`, 'success');
  updateStats();
  updateLineNumbers();
});

function doFind() {
  clearHighlights();
  const term = findInput.value;
  if (!term) { findCount.textContent = ''; return; }
  const text = editor.value;
  const lower = text.toLowerCase();
  const termLower = term.toLowerCase();
  findMatches = [];
  let idx = 0;
  while ((idx = lower.indexOf(termLower, idx)) !== -1) {
    findMatches.push(idx);
    idx += termLower.length;
  }
  findCount.textContent = findMatches.length ? `${findMatches.length} found` : 'No matches';
  if (findMatches.length > 0) {
    findIndex = 0;
    highlightMatch();
  }
}

function moveFindIndex(dir) {
  if (findMatches.length === 0) { doFind(); return; }
  findIndex = ((findIndex + dir) % findMatches.length + findMatches.length) % findMatches.length;
  highlightMatch();
}

function highlightMatch() {
  if (findMatches.length === 0) return;
  const pos = findMatches[findIndex];
  editor.focus();
  editor.setSelectionRange(pos, pos + findInput.value.length);
  findCount.textContent = `${findIndex + 1}/${findMatches.length}`;
  // Scroll into view
  const lines = editor.value.substring(0, pos).split('\n').length;
  const lineH = fontSize * 1.7;
  editor.scrollTop = Math.max(0, (lines - 3) * lineH);
}

function clearHighlights() {
  findMatches = [];
  findIndex = -1;
  findCount.textContent = '';
}

// ── Keyboard Shortcuts ──
document.addEventListener('keydown', e => {
  if (e.key === 'Escape') {
    const app = document.getElementById('app');
    if (app.classList.contains('fullscreen')) {
      app.classList.remove('fullscreen');
      $('btn-fullscreen').classList.remove('active');
      updateLineNumbers();
    }
    if (findBar.classList.contains('open')) {
      findBar.classList.remove('open');
      clearHighlights();
    }
    return;
  }
  if (!e.ctrlKey && !e.metaKey) return;
  switch (e.key.toLowerCase()) {
    case 's': e.preventDefault(); saveFile(); break;
    case 'f': e.preventDefault(); toggleFind(); break;
    case 'n': e.preventDefault(); newFile(); break;
    case 'o': e.preventDefault(); openFile(); break;
    case 'p': e.preventDefault(); printFile(); break;
  }
});

// ── Auto-save ──
function scheduleAutosave() {
  clearTimeout(autosaveTimer);
  autosaveTimer = setTimeout(autosave, 1500);
}

function autosave() {
  const tab = tabs.find(t => t.id === activeTabId);
  if (!tab) return;
  tab.content = editor.value;
  saveSession();
}

function saveSession() {
  try {
    const session = { tabs, activeTabId, untitledCount, fontSize };
    localStorage.setItem('np_session', JSON.stringify(session));
  } catch(e) {}
}

function restoreSession() {
  try {
    const raw = localStorage.getItem('np_session');
    if (!raw) return false;
    const session = JSON.parse(raw);
    if (!session.tabs || session.tabs.length === 0) return false;
    tabs = session.tabs;
    untitledCount = session.untitledCount || tabs.length;
    fontSize = session.fontSize || 14;
    document.documentElement.style.setProperty('--font-size', fontSize + 'px');
    $('font-size-display').textContent = fontSize;
    renderTabs();
    const id = tabs.find(t => t.id === session.activeTabId) ? session.activeTabId : tabs[0].id;
    switchTab(id);
    return true;
  } catch(e) { return false; }
}

// ── Theme restore ──
function restoreTheme() {
  const theme = localStorage.getItem('np_theme') || 'dark';
  document.documentElement.setAttribute('data-theme', theme);
  if (theme === 'light') {
    $('theme-icon-dark').style.display = 'none';
    $('theme-icon-light').style.display = '';
    $('theme-label').textContent = 'Dark';
  }
}

// ── Unsaved warning ──
window.addEventListener('beforeunload', e => {
  const unsaved = tabs.some(t => !t.saved);
  if (unsaved) { e.preventDefault(); e.returnValue = ''; }
  saveSession();
});

// ── Helpers ──
function escapeHtml(str) {
  return str.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;');
}

// ── Init ──
(function init() {
  restoreTheme();
  const restored = restoreSession();
  if (!restored) {
    newUntitled();
    toast('Welcome to Notepad ✦', 'info', 3000);
  } else {
    toast('Session restored', 'success');
  }
  $('btn-line-numbers').classList.add('active');
  updateStats();
  updateLineNumbers();
  updateCursor();
  editor.focus();

  // Sync line numbers scroll
  editor.addEventListener('scroll', () => {
    lineNums.scrollTop = editor.scrollTop;
  });

  // Periodic session save
  setInterval(saveSession, 30000);
})();
</script>
